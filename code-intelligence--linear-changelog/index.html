export function ExpiredInviteNotice({ invite }) {
  if (!invite || invite.status === "expired") {
    throw new InvalidInviteLinkError();
  }
  return validateInviteToken(invite.token);
}

async function redeemDriverInvite(token) {
  const result = await InviteTokenValidator.validate(token);
  if (result.expired) {
    await jobs.enqueue("send-driver-invite", {
      driverId: result.driverId,
      reason: "token_expired"
    });
  }
  return DriverOnboarding.resume(result.driverId);
}
